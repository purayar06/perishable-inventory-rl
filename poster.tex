% ---- Introduction ----
\begin{block}{Introduction \& Motivation}
\begin{itemize}
  \item Managing perishable goods (food, pharmaceuticals, blood products) poses a critical supply-chain challenge where \textbf{unsold items expire} and \textbf{unmet demand causes stockouts}.
  \item Ordering too much $\Rightarrow$ \textbf{waste} (expired units, financial loss).\\
        Ordering too little $\Rightarrow$ \textbf{stockouts} (lost sales, penalty costs).
  \item The optimal ordering policy must balance these \textbf{competing objectives} under \textbf{stochastic demand} - a sequential decision problem with \textbf{delayed consequences}.
  \item Classical heuristics (e.g., order-up-to) ignore the \textbf{age-structure} of inventory; RL methods can discover policies that condition on the full age profile.
\end{itemize}
\end{block}

% ---- Context ----
\begin{block}{Context: Demand \& Perishability}
\begin{itemize}
  \item \textbf{Demand:} $d_t \sim \text{Poisson}(\lambda)$, default $\lambda = 6$ units/day, truncated for DP.
  \item \textbf{Shelf-life:} $D$ days (default $D=3$); units age through $D$ age-buckets.
  \item \textbf{Selling rule:} FEFO (First-Expire-First-Out) --- oldest items sold first.
  \item \textbf{Episode horizon:} $T = 60$ days.
\end{itemize}
\end{block}

% ---- MDP Formulation ----
\begin{alertblock}{Problem Formulation (MDP)}

\textbf{State Space}
\[
  S_t = (n_1, n_2, \ldots, n_D), \quad n_i \in \{0, \ldots, N_{\max}\}
\]
$n_i$ = units with $i$ days of remaining shelf-life.\\
State-space size: $(N_{\max}+1)^{D}$.

\medskip
\textbf{Action Space}
\[
  a_t \in \{0, 1, \ldots, A_{\max}\}
\]

\textbf{Transition (one step)}
\begin{enumerate}
  \item Receive order: $n_D \leftarrow \min(n_D + a_t,\; N_{\max})$
  \item Sample demand: $d \sim \text{Poisson}(\lambda)$
  \item Sell FEFO: sell from oldest bucket first
  \item Waste: leftover in bucket 1 expires
  \item Age: $n_i \leftarrow n_{i+1}$, \; $n_D \leftarrow 0$
\end{enumerate}

\medskip
\textbf{Reward Function}
\[
  R_t = \underbrace{p \cdot \text{sold}}_{\text{revenue}}
      - \underbrace{c \cdot a_t}_{\text{order cost}}
      - \underbrace{w \cdot \text{waste}}_{\text{waste penalty}}
      - \underbrace{s \cdot \text{stockout}}_{\text{stockout penalty}}
\]

\begin{center}
\begin{tabular}{cccc}
\toprule
$p=10$ & $c=4$ & $w=6$ & $s=8$ \\
\bottomrule
\end{tabular}
\end{center}

\end{alertblock}

% ---- Methodology ----
\begin{block}{Methodology}
\begin{itemize}
  \item Custom \textbf{Gymnasium-style} simulation environment with \texttt{reset()}/\texttt{step()} API.
  \item Pure-function \texttt{simulate\_step()} for model-based DP; stochastic \texttt{step()} for model-free agents.
  \item Fixed-seed evaluation protocol (seed 123, 100 episodes) for fair comparison.
\end{itemize}
\end{block}

% ---- Agent Architectures ----
\begin{block}{Agent Architectures}
\begin{center}
\small
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Algorithm} & \textbf{Type} & \textbf{Update} \\
\midrule
DP -- Value Iter.      & Model-based   & $V(s)\!\leftarrow\!\max_a \sum_d P(d)[R\!+\!\gamma V(s')]$ \\[3pt]
Q-Learning             & Off-policy TD & $Q \leftarrow Q + \alpha[r\!+\!\gamma\max_{a'}Q' - Q]$ \\[3pt]
SARSA                  & On-policy TD  & $Q \leftarrow Q + \alpha[r\!+\!\gamma Q(s',a') - Q]$ \\[3pt]
Monte Carlo            & First-visit   & $Q \leftarrow Q + \alpha[G - Q]$ \\[3pt]
Linear FA              & Semi-grad.\ SARSA & $\mathbf{w}_a \leftarrow \mathbf{w}_a + \alpha\delta\boldsymbol{\phi}(s)$ \\
\bottomrule
\end{tabular}
\end{center}

% ---- Key Findings ----
\begin{alertblock}{Key Findings}
\begin{itemize}
  \item \textbf{DP} provides the theoretical upper bound but is intractable for large $D$ or $N_{\max}$ (state explosion $(N_{\max}\!+\!1)^D$).
  \item \textbf{Q-Learning \& SARSA} achieve near-optimal performance via tabular TD, converging within ${\sim}2\,000$ episodes.
  \item \textbf{SARSA} (on-policy) is more conservative: lower waste, slightly higher stockout --- reflecting the exploration--exploitation trade-off in ordering.
  \item \textbf{Monte Carlo} converges slower due to high variance over the 60-step horizon.
\end{itemize}
\end{alertblock}

\begin{itemize}
  \item Shorter shelf-life ($D\!=\!2$) $\Rightarrow$ significantly higher waste rates.
  \item Higher $\lambda$ stresses all agents $\Rightarrow$ increased waste \emph{and} stockout.
  \item Increasing $w$ drives under-ordering (less waste, more stockout).
  \item Increasing $s$ drives over-ordering (less stockout, more waste).
\end{itemize}

% ---- Conclusion ----
\begin{alertblock}{Conclusion \& Future Work}

\textbf{Conclusion}
\begin{itemize}
  \item \textbf{DP} yields the optimal baseline but scales as $\mathcal{O}\bigl((N_{\max}\!+\!1)^D\bigr)$.
  \item \textbf{Tabular TD} (Q-Learning, SARSA) reaches near-optimal performance without a model.
  \item \textbf{Linear FA} bridges tabular and deep RL with only $D\!+\!3$ hand-crafted features.
  \item The \textbf{waste--stockout trade-off} is the defining tension; on-policy methods favour safety.
\end{itemize}

\medskip
\textbf{Future Work}
\begin{enumerate}
  \item \textbf{Deep RL (DQN / PPO):} Neural function approximators for realistic $D \geq 7$ and continuous order quantities.
  \item \textbf{Non-stationary \& multi-product demand:} Time-varying distributions (seasonal), shared warehouse capacity --- closer to real retail.
\end{enumerate}

\end{alertblock}
